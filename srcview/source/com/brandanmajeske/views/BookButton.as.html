package com.brandanmajeske.views
{
    import com.brandanmajeske.vo.BookVO;
    
    import lib.ButtonBase;
    
    public class BookButton extends ButtonBase
    {
        private var _book:BookVO;
        
        public function BookButton()
        {
            super();
            mouseChildren = false;
            buttonMode = true;
        }

        public function get book():BookVO
        {
            return _book;
        }

        public function set book(value:BookVO):void
        {
            _book = value;
            tfLabel.text = _book.title;
        }

    }
}
