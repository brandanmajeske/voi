package com.brandanmajeske.book
{
    import com.brandanmajeske.vo.BookVO;
    
    import lib.BookViewBase1;
    
    public class BookView1 extends BookViewBase1 implements IBook
    {
        private var _book:BookVO;
        
        public function BookView1()
        {
            super();
        }
        
        public function updateUI():void
        {
            tfTitle.text = _book.title;
            tfAuthor.text = _book.author;    
            tfPublish.text = String(_book.publishDate);
            tfLanguage.text = _book.language;
            tfRating.text = _book.rating;
            tfCategory.text = _book.category;
        }
        
        
        public function resetUI():void
        {
            tfTitle.text = "No Title Selected";
            tfAuthor.text = "";    
            tfPublish.text = "";
            tfLanguage.text = "";
            tfRating.text = "";
            tfCategory.text = "";
        }

        public function get book():BookVO
        {
            return _book;
        }

        public function set book(value:BookVO):void
        {
            _book = value;
            updateUI();
        }

    }
}
