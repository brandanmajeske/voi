package
{
    import com.brandanmajeske.book.BookView1;
    import com.brandanmajeske.book.BookView2;
    import com.brandanmajeske.book.IBook;
    import com.brandanmajeske.ui.Button;
    import com.brandanmajeske.ui.LayoutBox;
    import com.brandanmajeske.views.BookButton;
    import com.brandanmajeske.views.ResetButton;
    import com.brandanmajeske.vo.BookVO;
    import flash.display.Sprite;
    import flash.events.MouseEvent;
    
    import lib.ButtonBase;
    
    [SWF("900", height="400", backgroundColor="#f5f5f5")]
    public class Main extends Sprite
    {
        private var _books:Array;
        private var _btnLb:LayoutBox;
        private var _viewLb:LayoutBox;
        private var _view1:IBook;
        private var _view2:IBook;
        
        
        public function Main()
        {
            _btnLb = new LayoutBox(5, true);    
            _viewLb = new LayoutBox(40, true);
            
            addChild(_btnLb);
            addChild(_viewLb);
            
            createBooks();
            createButtons();
            createView();
            
        }
        
        private function createView():void
        {
            
            _view1 = new BookView1();
            _view2 = new BookView2();
            
            _viewLb.addChild(BookView1(_view1));
            _viewLb.addChild(BookView2(_view2));
            _viewLb.x = 20;
            _viewLb.y = 100;
            
            _view1.resetUI();
            _view2.resetUI();
        }
        
        private function createBooks():void
        {
            
            var rc:BookVO = new BookVO();
            rc.title = "Robinson Crusoe";
            rc.author = "Daniel Defoe";
            rc.publishDate = 1719;
            rc.language = "English";
            rc.rating = "Must Read";
            rc.category = "Teen";
            
            var wp:BookVO = new BookVO();
            wp.title  = "War and Peace";
            wp.author = "Leo Tolstoy";
            wp.publishDate = 1869;
            wp.language = "Russian";
            wp.rating = "Very Long";
            wp.category = "Adults";
            
            var sf:BookVO = new BookVO();
            sf.title = "Superfudge";
            sf.author = "Judy Blume";
            sf.publishDate = 1980;
            sf.language = "English";
            sf.rating = "Entertaining";
            sf.category = "Children";
            
            var sd:BookVO = new BookVO();
            sd.title = "A Scanner Darkly";
            sd.author = "Philip K. Dick";
            sd.publishDate = 1977;
            sd.language = "English";
            sd.rating = "Trippy";
            sd.category = "Adults";
            
            var lx:BookVO = new BookVO();
            lx.title = "The Lorax";
            lx.author = "Dr. Seuss";
            lx.publishDate = 1972;
            lx.language = "English";
            lx.rating = "Classic";
            lx.category = "Children";
            
            _books = [rc, wp, sf, sd, lx];
        }
        
        private function createButtons():void
        {
            for each (var bk:BookVO in _books)
            {
                var btn:BookButton = new BookButton();
                btn.book = bk;
                _btnLb.addChild(btn);
                _btnLb.x = 12;
                _btnLb.y = 25;
                btn.addEventListener(MouseEvent.CLICK, btn_clickHandler);
                
            }
            
            var resetBtn:ResetButton = new ResetButton();
            resetBtn.resetBtn;
            _btnLb.addChild(resetBtn);
            resetBtn.addEventListener(MouseEvent.CLICK, resetBtn_clickHandler);
            
        }
        
        private function resetBtn_clickHandler(event:MouseEvent):void
        {
            var btn:ResetButton = ResetButton(event.currentTarget);
            _view1.resetUI();
            _view2.resetUI();
            
        }
        
        private function btn_clickHandler(event:MouseEvent):void
        {
            var btn:BookButton = BookButton(event.currentTarget);
            _view1.book = btn.book;
            _view2.book = btn.book;
            
        }        
    }
}
